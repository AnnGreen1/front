👇 查看代码注释，理解原型、原型链和构造函数

// 1. 构造函数（模具）
function Human(name, age) {
  this.name = name; // 张三独有的名字
  this.age = age; // 张三独有的年龄
}

// 2. 原型（公共工具箱）
// 我们把“走路”这个技能放在工具箱里，大家都共用
Human.prototype.walk = function() {
  console.log(this.name + " 正在用两条腿走路...");
};

// 3. 实例化（造人）
const zhangSan = new Human("张三", 20);
const liSi = new Human("李四", 25);

// --- 验证阶段 ---
// 4. 调用方法（去工具箱找技能）
zhangSan.walk(); // 输出：张三 正在用两条腿走路...
liSi.walk(); // 输出：李四 正在用两条腿走路...

// 5. 验证原型链（寻亲）
console.log("张三身上有 walk 吗？", zhangSan.hasOwnProperty('walk')); // false (自己没有)
console.log("张三的__proto__指向 Human.prototype 吗？", zhangSan.__proto__ === Human.prototype); // true (指引条正确)
console.log("Human.prototype 的__proto__指向 Object.prototype 吗？", Human.prototype.__proto__ === Object.prototype); // true (找到了祖宗)
