本地摄像头FLV流播放

使用Node.js捕获摄像头流并通过flv.js在浏览器中播放

摄像头播放器

准备就绪，请点击"连接摄像头流"按钮开始播放

使用说明

  1. 确保已安装Node.js和必要的依赖：npm install ws fluent-ffmpeg @ffmpeg-installer/ffmpeg @ffprobe-installer/ffprobe
  2. 运行服务端程序：npm start 或 node server.js
  3. 允许浏览器访问摄像头权限
  4. 点击上方"连接摄像头流"按钮开始播放

服务端代码示例

// server.js
const WebSocket = require('ws');
const ffmpeg = require('fluent-ffmpeg');
const { path: ffmpegPath } = require('@ffmpeg-installer/ffmpeg');

ffmpeg.setFfmpegPath(ffmpegPath);

// 创建WebSocket服务器
const wss = new WebSocket.Server({ port: 8081 });

wss.on('connection', (ws) => {
 // 启动摄像头流
 const stream = ffmpeg()
   .input('video=Integrated Camera') // Windows摄像头设备
   .inputFormat('dshow')
   .videoCodec('libx264')
   .size('1280x720')
   .fps(30)
   .outputFormat('flv')
   .on('data', (chunk) => {
     if (ws.readyState === WebSocket.OPEN) {
       ws.send(chunk, { binary: true });
     }
   })
   .run();
 
 ws.on('close', () => stream.kill());
});
